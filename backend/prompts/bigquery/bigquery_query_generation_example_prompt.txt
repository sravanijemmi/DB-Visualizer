# Note: In all examples, when a department is specified, all other filters (Account, Client Partner, Commercial BU, Deal Type, Deal Size, Customer Type, etc.) must be applied to both vw_budget and vw_opportunity as per the Filtering Logic. Only skip Department in vw_opportunity.

**Example-1**
--------------
**User Input**: What is XYZ BU budget achievement in April 2024 for SAP Department?

**AI Response**:
WITH budget AS (
  SELECT
    Commercial_BU_Name,
    SUM(CASE WHEN LOWER(TRIM(Department)) = 'sap' THEN COALESCE("April", 0) ELSE 0 END) AS budget_amount
  FROM `bcone-sales-insights-dev-prj.salesinsights_dev_bqd.vw_budget`
  WHERE LOWER(TRIM(Commercial_BU_Name)) = 'xyz'
    AND "Fiscal Year" = 'FY25'
  GROUP BY Commercial_BU_Name
),
opportunity AS (
  SELECT
    Commercial_BU,
    SUM(Amount_converted * (SAFE_CAST(SAP_Studio AS FLOAT64) / 100)) AS closed_amount
  FROM `bcone-sales-insights-dev-prj.salesinsights_dev_bqd.vw_opportunity`
  WHERE
    LOWER(TRIM(Commercial_BU)) = 'xyz'
    AND EXTRACT(MONTH FROM Close_Date) = 4
    AND EXTRACT(YEAR FROM Close_Date) = 2024
    AND Opportunity_Status = 'Closed'
    AND SAP_Studio IS NOT NULL
  GROUP BY Commercial_BU
)
SELECT
  b.Commercial_BU_Name,
  o.closed_amount,
  b.budget_amount,
  SAFE_DIVIDE(o.closed_amount, b.budget_amount) * 100 AS budget_achievement
FROM budget b
JOIN opportunity o
  ON b.Commercial_BU_Name = o.Commercial_BU;

**Example-2**
--------------
**User Input**: Year over Year (YoY) booking trend for Account 104 in Q1 FY25?

**AI Response**:

WITH bookings_fy24_q1 AS (
  SELECT
    SUM(Amount_converted) AS total_bookings
  FROM `bcone-sales-insights-dev-prj.salesinsights_dev_bqd.vw_opportunity`
  WHERE
    Account = 'Account 104'
    AND Close_Date >= DATE '2023-04-01'
    AND Close_Date < DATE '2023-07-01'
    AND Opportunity_Status = 'Closed'
),
bookings_fy25_q1 AS (
  SELECT
    SUM(Amount_converted) AS total_bookings
  FROM `bcone-sales-insights-dev-prj.salesinsights_dev_bqd.vw_opportunity`
  WHERE
    Account = 'Account 104'
    AND Close_Date >= DATE '2024-04-01'
    AND Close_Date < DATE '2024-07-01'
    AND Opportunity_Status = 'Closed'
)
SELECT
  b24.total_bookings AS bookings_fy24_q1,
  b25.total_bookings AS bookings_fy25_q1,
  SAFE_DIVIDE((b25.total_bookings - b24.total_bookings), b24.total_bookings) * 100 AS YoY_growth
FROM bookings_fy24_q1 b24
CROSS JOIN bookings_fy25_q1 b25;

WITH deals_fy24 AS (
  SELECT
    SAFE_DIVIDE(
      SUM(Amount_converted * (SAFE_CAST(SAP_Studio AS FLOAT64) / 100)),

    AND SAFE_CAST(SAP_Studio AS FLOAT64) > 0
),
deals_fy25 AS (
  SELECT
    SAFE_DIVIDE(
      SUM(Amount_converted * (SAFE_CAST(SAP_Studio AS FLOAT64) / 100)),
      COUNTIF(Opportunity_Status = 'Closed' AND SAFE_CAST(SAP_Studio AS FLOAT64) > 0)

**Example-6: Win Rate (Count-based)**
--------------
**User Input**: What is the win rate for Technology BU in Q1 FY25?

**AI Response**:
```sql
WITH opportunity_counts AS (
  SELECT
    COUNTIF(Opportunity_Status = 'Closed') AS won_opportunities,
    COUNTIF(Opportunity_Status IN ('Closed', 'Lost', 'Abandoned')) AS total_opportunities
  FROM `bcone-sales-insights-dev-prj.salesinsights_dev_bqd.vw_opportunity`
  WHERE
    LOWER(TRIM(Commercial_BU)) = 'technology'
    AND Close_Date >= DATE '2024-04-01'
    AND Close_Date < DATE '2024-07-01'  -- Q1 FY25: April-June 2024
)
SELECT
  won_opportunities,
  total_opportunities,
  SAFE_DIVIDE(won_opportunities, total_opportunities) * 100 AS win_rate
FROM opportunity_counts
WHERE total_opportunities > 0;
```

**Example-7: Win Rate with Department Weights**

```sql
WITH opportunity_counts AS (

  total_opportunities,
  SAFE_DIVIDE(won_opportunities, total_opportunities) * 100 AS win_rate
FROM opportunity_counts

```

**Example-10**
--------------
**User Input**: What is win ratio for XYZ BU in Q2 FY25 by deal size?

**AI Response**:

WITH opportunity AS (
  SELECT
    "Amount_converted",
    CASE WHEN "Amount_converted" >= 1000000 THEN 'Large' ELSE 'Small' END AS Deal_Size,
    "Opportunity_Status"
  FROM `bcone-sales-insights-dev-prj.salesinsights_dev_bqd.vw_opportunity`
  WHERE
    LOWER(TRIM("Commercial_BU")) = 'XYZ'
    AND "Opportunity_Status" IN ('Closed', 'Lost', 'Abandoned')
    AND "Close_Date" >= DATE '2024-07-01'
    AND "Close_Date" < DATE '2024-10-01'
)
SELECT
  Deal_Size,
  SUM(CASE WHEN "Opportunity_Status" = 'Closed' THEN "Amount_converted" ELSE 0 END) AS closed_amount,
  SUM("Amount_converted") AS total_amount,
  SAFE_DIVIDE(
    SUM(CASE WHEN "Opportunity_Status" = 'Closed' THEN "Amount_converted" ELSE 0 END),
    SUM("Amount_converted")
  ) * 100 AS win_ratio
FROM opportunity
GROUP BY Deal_Size
ORDER BY Deal_Size DESC;
