**Example-1**
--------------
**User Input**: What is the average age of opportunities for each department in the vw_opportunity table?
**AI Response**:
SELECT Department, AVG(age) AS Average_Age
FROM `bcone-sales-insights-dev-prj.salesinsights_dev_bqd.vw_opportunity`
GROUP BY Department;
 
**Example-2**
--------------
**User Input**: What is the month-over-month growth rate of the budget for each department in the vw_budget table in FY25?
**AI Response**:
WITH budget AS (
  SELECT
    Department,
    SUM(COALESCE("April", 0)) AS April,
    SUM(COALESCE("May", 0)) AS May,
    SUM(COALESCE("June", 0)) AS June,
    SUM(COALESCE("July", 0)) AS July,
    SUM(COALESCE("August", 0)) AS August,
    SUM(COALESCE("September", 0)) AS September,
    SUM(COALESCE("October", 0)) AS October,
    SUM(COALESCE("November", 0)) AS November,
    SUM(COALESCE("December", 0)) AS December,
    SUM(COALESCE("January", 0)) AS January,
    SUM(COALESCE("February", 0)) AS February,
    SUM(COALESCE("March", 0)) AS March
  FROM `bcone-sales-insights-dev-prj.salesinsights_dev_bqd.vw_budget`
  WHERE "Fiscal Year" = 'FY25'
  GROUP BY Department
)
SELECT
  Department,
  ((May - April) / NULLIF( April, 0 )) * 100 AS April_May_growth,
  ((June - May) / NULLIF( May, 0 )) * 100 AS May_June_growth,
  ((July - June) / NULLIF( June, 0 )) * 100 AS June_July_growth,
  ((August - July) / NULLIF( July, 0 )) * 100 AS July_August_growth,
  ((September - August) / NULLIF( August, 0 )) * 100 AS August_September_growth,
  ((October - September) / NULLIF( September, 0 )) * 100 AS September_October_growth,
  ((November - October) / NULLIF( October, 0 )) * 100 AS October_November_growth,
  ((December - November) / NULLIF( November, 0 )) * 100 AS November_December_growth,
  ((January - December) / NULLIF( December, 0 )) * 100 AS December_January_growth,
  ((February - January) / NULLIF( January, 0 )) * 100 AS January_February_growth,
  ((March - February) / NULLIF( February, 0 )) * 100 AS February_March_growth
FROM budget;
 
**Example-3**
--------------
**User Input**: What is the percentage of the total closed amount that each department contributes in FY25?
**AI Response**:
WITH total_closed_amount AS (
  SELECT
    SUM(Amount_converted) AS total_amount
  FROM `bcone-sales-insights-dev-prj.salesinsights_dev_bqd.vw_opportunity`
  WHERE
    Close_Date >= DATE '2024-04-01'
    AND Close_Date < DATE '2025-04-01'
    AND Opportunity_Status = 'Closed'
),
department_closed_amount AS (
  SELECT
    Department,
    SUM(Amount_converted) AS department_amount
  FROM `bcone-sales-insights-dev-prj.salesinsights_dev_bqd.vw_opportunity`
  WHERE
    Close_Date >= DATE '2024-04-01'
    AND Close_Date < DATE '2025-04-01'
    AND Opportunity_Status = 'Closed'
  GROUP BY Department
)
SELECT
  d.Department,
  SAFE_DIVIDE(d.department_amount, t.total_amount) * 100 AS department_contribution_percentage
FROM department_closed_amount d
CROSS JOIN total_closed_amount t;
 
**Example-4**
--------------
**User Input**: What is the average deal size for Kinaxis Department YTD?
**AI Response**:
SELECT
  SUM(Amount_converted) / COUNT(*) AS average_deal_size
FROM `bcone-sales-insights-dev-prj.salesinsights_dev_bqd.vw_opportunity`
WHERE
  LOWER(TRIM(Department)) = 'kinaxis'
  AND Opportunity_Status = 'Closed'
  AND Close_Date >= DATE_TRUNC(CURRENT_DATE(), YEAR)
  AND Close_Date <= CURRENT_DATE();
 
**Example-5**
--------------
**User Input**: What is average deal velocity for Account 110 in December 2024?
**AI Response**:
SELECT Account, AVG(DATE_DIFF(Close_Date, CAST(Created_Date AS DATE), DAY)) AS average_deal_velocity
FROM `bcone-sales-insights-dev-prj.salesinsights_dev_bqd.vw_opportunity`
WHERE Account = 'Account 110'
  AND EXTRACT(MONTH FROM Close_Date) = 12
  AND EXTRACT(YEAR FROM Close_Date) = 2024
  AND Opportunity_Status = 'Closed'
GROUP BY Account;
